# maven项目打包成可执行jar文件配置
---

## pom.xml里添加如下内容

```xml
<build>
      <!--打包生成的最终名称-->
      <finalName>mydemo</finalName>
      <pluginManagement>          
              <plugin>
                  <artifactId>maven-jar-plugin</artifactId>
                  <version>3.0.2</version>
                  <configuration>
                      <archive>
                          <manifest>
                              <!--类路径（我也不懂）-->
                              <addClasspath>true</addClasspath>
                              <!--包路径（我也不懂）-->
                              <classpathPrefix>lib</classpathPrefix>
                              <!--主类-->
                              <mainClass>
                                  com.zhou.webofscience.patent.MyFrame
                              </mainClass>
                          </manifest>
                      </archive>
                  </configuration>
              </plugin>              
          </plugins>
      </pluginManagement>
  </build>
```
